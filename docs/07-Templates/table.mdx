```tsx live noInline
interface Props {
  query: string;
  changeQuery: (query: string) => void;
  changeFilter: (filter: { name: string; value: any }) => void;
  filters: any[];
}

const TablePage = ({ changeQuery, query, changeFilter, filters }: Props) => {
  const columns = useColumns();

  return (
    <Page navId="global-users">
      <Page.Contents>
        <Flex gap={2}>
          <Box display={'flex'} gap={2}>
            <FilterInput
              placeholder="Search user by login, email, or name."
              autoFocus={true}
              value={query}
              onChange={changeQuery}
            />
            <RadioButtonGroup
              options={[
                { label: 'All users', value: false },
                { label: 'Active last 30 days', value: true },
              ]}
              onChange={(value) => changeFilter({ name: 'activeLast30Days', value })}
              value={filters.find((f) => f.name === 'activeLast30Days')?.value}
            />
            <LinkButton href="admin/users/create" variant="primary">
              New user
            </LinkButton>
          </Box>
          <InteractiveTable columns={columns} data={users} getRowId={(user) => String(user.id)} />
        </Flex>
      </Page.Contents>
    </Page>
  );
};

render(<TablePage query={''} changeQuery={() => {}} changeFilter={() => {}} filters={[]} />);
```
